18:  R[25]=0    ;
19:  LBL[1] ;
20:  SELECT R[25]=1,JMP LBL[2] ;

28:  JMP LBL[1] ;
29:  ENDIF ;
30:   ;

40:   ;
41:  LBL[2] ;

44:  PR[10,6]=PR[10,6]+R[11:D_ANGLE]    ;
45:L PR[10] 100mm/sec FINE    ;
46:  PR[11]=LPOS    ;
47:  PR[11,2]=R[19]    ;
48:L PR[11] 100mm/sec FINE    ;
49:  R[25]=0    ;

51:   ;

53:  R[12]=R[2:ANA_VAL]-R[5:ANA_VAL_Z2]    ;

55:  PR[10,6]=PR[10,6]-R[11:D_ANGLE]    ;
56:L PR[10] 100mm/sec FINE    ;
57:  PR[11]=LPOS    ;
